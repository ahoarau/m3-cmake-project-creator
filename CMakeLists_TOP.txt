cmake_minimum_required(VERSION 2.8)
project(__PROJECT_CREATOR_PROJECT_NAME__)

# Set the __PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT variable
set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT ${PROJECT_SOURCE_DIR})
# Set where to find the FindXXX.cmake
set(CMAKE_MODULE_PATH "$ENV{M3_ROBOT}/../meka-cmake-modules/Modules/" ${CMAKE_MODULE_PATH})

# Set all bunch of usefull variables
set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___INCLUDE_DIR ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/include)

set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___PROTO_DIR ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/proto)

set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___SRC_DIR ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/src)

set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___INSTALL_BIN ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/bin)

set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___BINARY_OUTPUT_DIRECTORY ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/bin)

set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___INSTALL_LIB ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/lib)

set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___LIBRARY_OUTPUT_DIRECTORY ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/lib)

set(__PROJECT_CREATOR_PROJECT_NAME_UPPER___PYTHON_DIR ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___ROOT}/python)

# Forcing the generation dirs to be local
set(CMAKE_BINARY_OUTPUT_DIRECTORY  ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___BINARY_OUTPUT_DIRECTORY})
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___LIBRARY_OUTPUT_DIRECTORY})

include_directories(
    ${__PROJECT_CREATOR_PROJECT_NAME_UPPER___INCLUDE_DIR}
    ${CMAKE_BINARY_DIR}/src # Include this directory for the generated protobufs headers
)


# Add the main project src dir that contains all the others
add_subdirectory(src/__PROJECT_CREATOR_PROJECT_NAME__)
